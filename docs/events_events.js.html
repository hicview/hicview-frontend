'use strict'

/**
 * HiC Event
 *
 * @class HiCEvent
 */
class HiCEvent {

  /**
   *Creates an instance of HiCEvent.
   * @memberof HiCEvent
   */
  constructor () {

  }

  /**
   * Add a {name:val} to HiC Event
   *
   * @param {String} name
   * @param {*} val
   * @returns {HiCEvent}
   * @memberof HiCEvent
   */
  attr (name, val) {
    if (typeof val === 'object') {
      this[name] = Object.assign(val)
    } else {
      this[name] = val
    }

    return this
  }
}

export { HiCEvent }
